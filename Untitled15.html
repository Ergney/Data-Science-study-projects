              for ii in range(len(W)):
                  count += 1

                  if L1 == True:
                      coef = lmbd * sum(abs(W))/2

      def eval_sgd_model(data, target, iterations=1000, alpha=0.0001, lmbd=1e-2, L1=False, L2=False):
          result_list_mse = []
          result_list_W = []
          W = np.ones(data[0].shape, dtype=float)
          n = len(data)
          count = 0
          for i in range(iterations):
              for j, k in itertools.product(range(n), range(len(W))):
                  count += 1
                  coef = 0
                  y_pred = np.dot(data, W)
                  if L1 == True:
                      coef = lmbd * sum(abs(W))/2
                  elif L2 == True:
                      coef = lmbd * sum(W**2)/2
                  W[k] -= 2 * alpha * np.dot(data[j].T, (np.dot(data[j], W[k]) - target[j])) + coef / n
                  if count % len(W) == 0:
                      error = calc_mse(target, y_pred)
                                            
              result_list_mse.append(error)
              result_list_W.append(W)
          return result_list_mse, result_list_W.append, count

      def norm(data):
          n1, n2  = data.shape[0], data.shape[1]
           
          data = data.T
          for i in range(n2):
              min_data = data[i].min()
              max_data = data[i].max()
              for j in range(n1):
                  data[i][j] = (data[i][j] - min_data)/(max_data - min_data)
                  
          return data.T

      def standart(data):
          n1, n2  = data.shape[0], data.shape[1]
          
          data = data.T
          for i in range(n2):
              mean = np.mean(data[i])
              stdev = np.sqrt(sum((data[i]-mean)**2)/n1)
              
              for j in range(n1):
                  data[i][j] = (data[i][j] - mean)/stdev
          
          return data.T
